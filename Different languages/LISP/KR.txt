(DEFUN KR (Q M1 N1)
 (PROG (X M N V X1)
   (SETQ X Q)
   (SETQ X1 Q)
   (SETQ M M1)
   (SETQ N N1)
   (SETQ V 0)
   (COND ( (NULL X) (RETURN NIL) )
         ( (< (LENGTH X) N) (RETURN NIL))
         ( (> M N) (RETURN NIL) )
         ( (< M 1) (RETURN NIL) )
   )
   A 
   (SETQ V (+ 1 V) )
   (COND ( (< V M) (DELETE (CAR X1) X) )
         ( (> V N) (DELETE (CAR X1) X) )        
   )
   (SETQ X1 (CDR X1))
   (COND ((= (LENGTH X1) 0) (RETURN X)))  
   (GO A)
 )
)
